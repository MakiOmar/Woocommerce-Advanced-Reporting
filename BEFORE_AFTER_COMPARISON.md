#### AFTER (Refactored):
```php
/**
 * AJAX Handler: Product/Customer Live Search
 *
 * @since 6.1
 * @return void
 */
add_action( 'wp_ajax_pw_chosen_ajax', 'pw_chosen_ajax' );  // ‚úÖ Only authenticated
function pw_chosen_ajax() {
	// Verify nonce
	if ( ! PW_Report_AWR_Helpers::verify_nonce( 'nonce', 'pw_livesearch_nonce' ) ) {  // ‚úÖ Helper method
		PW_Report_AWR_Helpers::json_error( 'Invalid security token', 403 );  // ‚úÖ Standardized error
	}

	// Check capabilities
	if ( ! PW_Report_AWR_Helpers::current_user_can( 'manage_woocommerce' ) ) {  // ‚úÖ Authorization
		PW_Report_AWR_Helpers::json_error( 'Insufficient permissions', 403 );
	}

	// Sanitize and parse input
	$postdata = isset( $_REQUEST['postdata'] ) ? sanitize_text_field( wp_unslash( $_REQUEST['postdata'] ) ) : '';  // ‚úÖ Sanitized
	parse_str( $postdata, $my_array_of_vars );

	$target_type  = isset( $my_array_of_vars['target_type'] ) ? sanitize_text_field( $my_array_of_vars['target_type'] ) : '';  // ‚úÖ Sanitized
	$search_query = isset( $my_array_of_vars['q'] ) ? sanitize_text_field( $my_array_of_vars['q'] ) : '';  // ‚úÖ Sanitized

	// Validate target type
	$allowed_types = array( 'simple_products', 'variable_products', 'all_products', 'customer' );  // ‚úÖ Whitelist
	if ( ! in_array( $target_type, $allowed_types, true ) ) {  // ‚úÖ Validation
		PW_Report_AWR_Helpers::json_error( 'Invalid target type', 400 );
	}

	global $pw_rpt_main_class;
	$data = array();

	switch ( $target_type ) {
		case 'simple_products':
			$data = $pw_rpt_main_class->pw_get_product_woo_data_chosen( 'simple', false, $search_query );
			break;
		// ... more cases
	}

	PW_Report_AWR_Helpers::json_success( $data );  // ‚úÖ Standardized success
}
```
